## Usage

<Subtitle>直接导入使用 ( 需要指明当前 `ThemeMode` )</Subtitle>

<Source
  language="tsx"
  dark
  code={`
"use client";

import { SHADOW_STYLES } from "venomous-ui-react/constants";

export default function App() {
  console.log(SHADOW_STYLES.light[1], SHADOW_STYLES.dark[1], SHADOW_STYLES.dark[2]);
  return null;
}
  `.trim()}
/>

<Subtitle>使用 `useThemeDesigns()` 获取</Subtitle>

<Source
  language="tsx"
  dark
  code={`
"use client";

import { Theme } from "venomous-ui-react/components";
import { useThemeDesign } from "venomous-ui-react/hooks";

export function App() {
  return (
    <Theme.Provider>
      <Component />
    </Theme.Provider>
  );
}

export function Component() {
  const { ShadowStyles } = useThemeDesigns();
  return (
    <div style={{ boxShadow: ShadowStyles[1] }}>
      Hello World
    </div>
  )
}
`.trim()}

/>
