## Usage

<Subtitle>直接导入使用 ( 需要指明当前 `ThemeMode` )</Subtitle>

<Source
  language="tsx"
  dark
  code={`
"use client";

import { BACKGROUND_COLORS } from "venomous-ui-react/constants";

export default function App() {
  console.log(BACKGROUND_COLORS.light[1], BACKGROUND_COLORS.dark[1], BACKGROUND_COLORS.dark[2]);
  return null;
}
  `.trim()}
/>

<Subtitle>使用 `useThemeDesigns()` 获取</Subtitle>

<Source
  language="tsx"
  dark
  code={`
"use client";

import { Theme } from "venomous-ui-react/components";
import { useThemeDesign } from "venomous-ui-react/hooks";

export function App() {
  return (
    <Theme.Provider>
      <Component />
    </Theme.Provider>
  );
}

export function Component() {
  const { BackgroundColors } = useThemeDesigns();
  return (
    <div style={{ backgroundColor: BackgroundColors[1] }}>
      Hello World
    </div>
  )
}
`.trim()}
/>
