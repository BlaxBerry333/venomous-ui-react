# useElementHoverEvents

用于管理元素的 mouse hover 状态和事件处理。

<Source
  language="tsx"
  dark
  code={`
"use client";

import { useElementHoverEvents } from "@/venomous-ui-react/hooks";
`.trim()}
/>

## Types

<Source
  language="ts"
  dark
  code={`
type HookParametersType = {
  disabled?: boolean;                          // 是否禁用 hover 效果
  onMouseEnter?: React.MouseEventHandler<T>;   // 外部传入的 mouseEnter 事件
  onMouseLeave?: React.MouseEventHandler<T>;   // 外部传入的 mouseLeave 事件
  onLeaveCallback?: () => void;                // hover 离开时的额外回调
}
type HookReturnType = {
  isHovered: boolean;                          // 当前是否处于 hover 状态
  MouseEnterEvent: React.MouseEventHandler<T>; // 合并后的 mouseEnter 事件
  MouseLeaveEvent: React.MouseEventHandler<T>; // 合并后的 mouseLeave 事件
}
`.trim()}
/>

## Usage

<Source
  language="tsx"
  dark
  code={`
"use client";

import React from "react";
import { useElementHoverEvents } from "@/venomous-ui-react/hooks";

export function App() {
  return (
    <Theme.Provider>
      <Component />
    </Theme.Provider>
  );
}

export function Component() {
  const { isHovered, MouseEnterEvent, MouseLeaveEvent } = useElementHoverEvents({
    disabled: false,
    onMouseEnter: e => console.log(e),
    onMouseLeave e => console.log(e),
  });
  
  return (
    <div
      onMouseEnter={MouseEnterEvent}
      onMouseLeave={MouseLeaveEvent}
      style={{ backgroundColor: isHovered ? "#e0e0e0" : "#f5f5f5" }}
    >
      Content
    </div>
  ); 
} `.trim()} />
